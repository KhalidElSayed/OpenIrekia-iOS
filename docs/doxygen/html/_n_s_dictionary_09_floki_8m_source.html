00072             case '?': [buf appendString:@"?"]; break;
00073             case ':': [buf appendString:@":"]; break;

00078             case '+': [buf appendString:@"+"]; break;
00079             case '.': [buf appendString:@"."]; break;
00080             case ',': [buf appendString:@","]; break;
00081             case '%': [buf appendString:@"%"]; break;
00082             default:
