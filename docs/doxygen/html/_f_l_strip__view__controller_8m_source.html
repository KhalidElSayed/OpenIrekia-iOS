00662 - (void)willRotateToInterfaceOrientation:
00663     (UIInterfaceOrientation)toInterfaceOrientation
00664     duration:(NSTimeInterval)duration
00665 {
00666     ignore_scrolls_ = YES;
00667     DLOG(@"Rotating! Weeeeee, we were looking at page %d", page_);
00668 
00669     const BOOL to_landscape =
00670         (UIInterfaceOrientationLandscapeRight == toInterfaceOrientation ||
00671             UIInterfaceOrientationLandscapeLeft == toInterfaceOrientation);
00672     if (IS_IPAD) {
00673         /* iPads have a non full screen view, they require special sizes. */
00674         if (to_landscape) {
00675             size_.width = 1024 - 320;
00676             size_.height = 768;
00677         } else {
00678             size_.width = 768;
00679             size_.height = 1024;
00680         }
00681         const CGRect rect = [[UIApplication sharedApplication] statusBarFrame];
00682         const CGFloat offset = MIN(rect.size.width, rect.size.height);
00683         size_.height -= offset;
00684     } else {
00685         // Detect and store the new size.
00686         size_ = [self get_ipad_visible_area];
00687         BOOL swap = NO;
00688         if (to_landscape) {
