00432 - (void)did_receive_beacon_data:(id)response error:(NSError*)error
00433 {
00434     if (error) {
00435         [self handle_error:_e(4) with_error:error abort_if_needed:YES];
00436         // _4: Error getting app beacon
00437         return;
00438     }
00439     // _
00440 
00441     NSString *response_string = [[[NSString alloc]
00442         initWithData:[connection_ data] encoding:NSUTF8StringEncoding]
00443         autorelease];
00444 
00445     NSDictionary *result = [response_string JSONValue];
00446     if (!result) {
00447         [self handle_error:_e(5) with_message:response_string
00448             abort_if_needed:YES];
00449         // _5: Invalid app beacon JSON
00450         return;
00451     }
00452 
00453     static NSString *required_keys[] = {@"url", @"v", @"ttl", 0};
00454     for (int f = 0; required_keys[f]; f++) {
00455         if (![result objectForKey:required_keys[f]]) {
00456             [self handle_error:_e(6) with_message:required_keys[f]
00457                 abort_if_needed:YES];
00458             // _6: Missing beacon key
00459             return;
00460         }
00461     }
00462 
00463     /* Store the beacon string for saving later. We don't want
00464      * to save the beacon with the new app data version until the
00465      * real app data has been downloaded. Otherwise a cache
00466      * inconsistency could be generated and the world would implode
00467      * into itself. Oh the horror!
00468      */
00469     [beacon_string_ release];
00470     beacon_string_ = [response_string retain];
00471 
00472     BOOL fetch_data = NO;
00473     if (app_is_initialised_) {
00474         const int online_version = [result get_int:@"v" def:-1];
00475         if (beacon_cached_version_ != online_version) {
00476             DLOG(@"Version of data changed, refetching.");
00477             fetch_data = YES;
00478         } else {
00479             /* Everything OK. Refresh the beacon string and app data files. */
00480             const char *path = [get_path(FILE_APP_DATA, DIR_CACHE)
00481                 cStringUsingEncoding:1];
00482             _update_timestamp(path);
00483 
00484             if (![self save_as_beacon:response_string])
00485                 return;
00486 
00487             [self performSelector:@selector(check_cached_data)
00488                 withObject:nil afterDelay:CHECK_CACHED_DATA_DELAY];
00489         }
00490     } else {
00491         DLOG(@"Fetching data application because its the first time.");
00492         fetch_data = YES;
00493     }
00494 
00495     if (fetch_data) {
00496         [connection_ release];
00497         connection_ = [[FLRemote_connection alloc]
00498             init_with_action:@selector(did_receive_app_data:error:)
00499             target:self];

00501             prettify_request_url:[result objectForKey:@"url"]]];
